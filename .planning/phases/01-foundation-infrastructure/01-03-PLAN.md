    **.rnstorybook/preview.tsx** (font decorator — non-negotiable per CONTEXT.md):
    ```typescript
    import React from "react";
    import { View, ActivityIndicator } from "react-native";
    import { useFonts } from "expo-font";
    import { DMSans_400Regular, DMSans_500Medium } from "@expo-google-fonts/dm-sans";
    import { DMMono_400Regular } from "@expo-google-fonts/dm-mono";
    import type { Preview } from "@storybook/react-native";
    import "../global.css"; // NativeWind requires this import in Storybook context — separate from app

    const preview: Preview = {
      decorators: [
        (Story) => {
          // Font loading guard: stories render in system font until fonts load
          // This decorator ensures all stories render with Morph Maternity brand fonts
          const [fontsLoaded, fontError] = useFonts({
            // Cormorant Garamond: local .ttf files (see assets/fonts/INSTRUCTIONS.md)
            "CormorantGaramond-Regular": require("../assets/fonts/CormorantGaramond-Regular.ttf"),
            "CormorantGaramond-Italic": require("../assets/fonts/CormorantGaramond-Italic.ttf"),
            "CormorantGaramond-SemiBold": require("../assets/fonts/CormorantGaramond-SemiBold.ttf"),
            "CormorantGaramond-Bold": require("../assets/fonts/CormorantGaramond-Bold.ttf"),
            // DM Sans: via @expo-google-fonts package
            "DMSans-Regular": DMSans_400Regular,
            "DMSans-Medium": DMSans_500Medium,
            // DM Mono: via @expo-google-fonts package
            "DMMono-Regular": DMMono_400Regular,
          });

          if (!fontsLoaded && !fontError) {
            // Show loading indicator until fonts are ready — prevents false sign-off on system font
            return (
              <View className="flex-1 items-center justify-center bg-cream-50">
                <ActivityIndicator />
              </View>
            );
          }

          return (
            <View className="flex-1 bg-cream-50 p-4">
              <Story />
            </View>
          );
        },
      ],
      parameters: {
        controls: {
          matchers: {
            color: /(background|color)$/i,
            date: /Date$/,
          },
        },
      },
    };

    export default preview;
    ```

    IMPORTANT: The `require("../assets/fonts/CormorantGaramond-Regular.ttf")` calls require the .ttf files to exist. If they are not present yet (user has not downloaded them), wrap in try/catch as shown in Plan 01 Task 2. The checkpoint in Task 3 of this plan will verify fonts render correctly once files are placed.
